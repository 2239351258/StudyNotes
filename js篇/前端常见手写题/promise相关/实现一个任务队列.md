### 题一

#### 题意：

设计一个简单的任务队列, 要求分别在 1,3,4 秒后打印出 "1", "2", "3"

```js
new Quene()
  .task(1000, () => {
  console.log(1)
})
  .task(2000, () => {
  console.log(2)
})
  .task(1000, () => {
  console.log(3)
})
  .start()

function Quene() { ... } //补全代码
```

#### 实现

```js
class Quene{
  taskQuene = []
  construction() { }
  task(delay, cb) {
    this.taskQuene.push({delay,cb})
    return this
  }
  start() {
    let next = Promise.resolve()
    this.taskQuene.forEach((item) => {
      const { delay, cb } = item
      next = next.then(() => {
        return new Promise((res, rej) => {
          setTimeout(() => {
            cb()
            res()
          }, delay)
        })
      })
    })
  }
}
```
